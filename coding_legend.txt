#include <sstream>
#include <vector>
#include <iostream>
#include <string>
#include <map>
#include <algorithm>
#include <math.h>
#include <set>
using namespace std;



void solve(){
    int t;
    vector<vector<int>> bils;
    cin >> t;
    for(int i = 0;i < t;i++){
        vector<int> bil;
        string sbil;
        cin >> sbil;
        for(char b : sbil){
            bil.push_back(b - '0');
        }
        bils.push_back(bil);
    }
    for(vector<int> bil : bils){
        vector<int> ans;
        int max;
        for(int i = 0;i < bil.size();i++){
            if(bil[i] <= 0)
                continue;
            ans.push_back(bil[i] * pow(10,bil.size() - (i + 1)));
        }
        max = *max_element(ans.begin(),ans.end());
        sort(ans.begin(),ans.end(),greater<int>());
        cout << ans.size() << endl;
        if(ans.size() == 2){
            cout << ans[0] << " " << ans[1] << endl;
            continue;
        }
        for(int val : ans){
            if(val == max)
                continue;
            cout << val << " ";
        }
        cout << max;
        cout << endl;
    }
}

int main() {
    solve();
	return 0;
}
